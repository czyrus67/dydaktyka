    <div id="wynik"></div>

    <script>
        function suma_sta() {
            var a = document.getElementById("a").value;
            var b = document.getElementById("b").value;

            if (a == "" || b == "") {
                document.getElementById("wynik").innerHTML = "Proszę uzupełnić obie liczby.";
            } else {
                a = parseFloat(a);
                b = parseFloat(b);
                var suma = a + b;
                document.getElementById("wynik").innerHTML = "Wynik działania wynosi: " + suma.toFixed(2);
            }
        }

        function roznica_sta() {
            var a = document.getElementById("a").value;
            var b = document.getElementById("b").value;

            if (a == "" || b == "") {
                document.getElementById("wynik").innerHTML = "Proszę uzupełnić obie liczby.";
            } else {
                a = parseFloat(a);
                b = parseFloat(b);
                var roznica = a - b;
                document.getElementById("wynik").innerHTML = "Wynik działania wynosi: " + roznica.toFixed(2);
            }
        }

        function iloczyn_sta() {
            var a = document.getElementById("a").value;
            var b = document.getElementById("b").value;

            if (a == "" || b == "") {
                document.getElementById("wynik").innerHTML = "Proszę uzupełnić obie liczby.";
            } else {
                a = parseFloat(a);
                b = parseFloat(b);
                var iloczyn = a * b;
                document.getElementById("wynik").innerHTML = "Wynik działania wynosi: " + iloczyn.toFixed(2);
            }
        }

        function iloraz_sta() {
            var a = document.getElementById("a").value;
            var b = document.getElementById("b").value;

            if (a == "" || b == "") {
                document.getElementById("wynik").innerHTML = "Proszę uzupełnić obie liczby.";
            } else if (b == 0) {
                document.getElementById("wynik").innerHTML = "Nie wolno dzielić przez zero.";
            } else {
                a = parseFloat(a);
                b = parseFloat(b);
                var iloraz = a / b;
                document.getElementById("wynik").innerHTML = "Wynik działania wynosi: " + iloraz.toFixed(2);
            }
        }
