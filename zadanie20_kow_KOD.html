function AB_kry() {
        var xa_kow = parseFloat(document.getElementById('xa').value);
        var ya_kow = parseFloat(document.getElementById('ya').value);
        var xb_kow = parseFloat(document.getElementById('xb').value);
        var yb_kow = parseFloat(document.getElementById('yb').value);
        var ab = Math.sqrt(Math.pow(xb_kow - xa_kow, 2) + Math.pow(yb_kow - ya_kow, 2));
        document.getElementById("wynik").innerHTML = "Długość AB = " + ab.toFixed(2) + " cm";
    }

    function BC_kry() {
        var xb_kow = parseFloat(document.getElementById('xb').value);
        var yb_kow = parseFloat(document.getElementById('yb').value);
        var xc_kow = parseFloat(document.getElementById('xc').value);
        var yc_kow = parseFloat(document.getElementById('yc').value);
        var bc = Math.sqrt(Math.pow(xc_kow - xb_kow, 2) + Math.pow(yc_kow - yb_kow, 2));
        document.getElementById("wynik").innerHTML = "Długość BC = " + bc.toFixed(2) + " cm";
    }
