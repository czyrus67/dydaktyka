    <script language="JavaScript">
        function fun_kow() {
            var opcja_dana_kow = prompt("Podaj kolor (1, 'czerwony', 2 lub 'niebieski'):", "").toLowerCase();

            if (opcja_dana_kow == "1" || opcja_dana_kow == "czerwony") {
                document.write('<h1 style="color: red; font-size: 7;">Wczytałeś kolor czerwony</h1><br />');
            } else if (opcja_dana_kow == "2" || opcja_dana_kow == "niebieski") {
                document.write('<h6 style="color: blue; font-size: 1;">Wczytałeś kolor niebieski</h6><br />');
            } else {
                document.write('<p style="color: black; font-size: 3;">Niepoprawny wybór!</p>');
            }
        }
    </script>
