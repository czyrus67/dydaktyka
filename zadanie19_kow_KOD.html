function pole_kowalski() {
        var a_kow = parseFloat(document.getElementById('dlugosc').value);
        var b_kow = a_kow * a_kow;
        document.getElementById("nazwa").innerHTML = "Pole kwadratu= " + b_kow.toFixed(2) + " cm²";
    }

    function obwod_kowalski() {
        var a_kow = parseFloat(document.getElementById('dlugosc').value);
        var b_kow = 4 * a_kow;
        document.getElementById("nazwa").innerHTML = "Obwód kwadratu= " + b_kow.toFixed(2) + " cm";
    }
