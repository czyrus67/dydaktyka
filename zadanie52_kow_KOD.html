<script type="text/javascript">
    // Dane osobowe
    var nazwisko = "Starkowicz";
    var imie = "Michał";

    // Obliczanie a i b
    var a_kow = ((nazwisko.length) % 4) + 5; // a = ((liczba_liter_nazwiska) mod 4) + 5
    var b_kow = ((imie.length) % 6) + 20;    // b = ((liczba_liter_imienia) mod 6) + 20

    // Górna linia gwiazdek
    for (var i = 0; i < b_kow; i++) {
        document.write("*");
    }
    document.write("<br>");

    // Środkowe linie z pionowymi gwiazdkami
    for (var w = 0; w < a_kow - 2; w++) {
        document.write("*");
        for (var j = 0; j < b_kow - 2; j++) {
            document.write(" ");
        }
        document.write("*<br>");
    }

    // Dolna linia gwiazdek
    for (var i = 0; i < b_kow; i++) {
        document.write("*");
    }
    document.write("<br><br>");
    document.write("a = " + a_kow + "<br>");
    document.write("b = " + b_kow + "<br>");
</script>
