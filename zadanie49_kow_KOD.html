        <script type="text/javascript">
            var n_kow = parseInt(prompt("Podaj liczbę n (0-20):", ""));
            var silnia_kow = 1;
            if (n_kow === 0) {
                silnia_kow = 1;
            } else {
                for (var i = 1; i <= n_kow; i++) {
                    silnia_kow *= i;
                }
            }
            document.write("n=" + n_kow + "<br>silnia=" + silnia_kow);
        </script>
