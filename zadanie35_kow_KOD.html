        // Wczytanie liczby od użytkownika
        var kow = prompt("Podaj liczbę:", "");
        var a_dana_kow = parseFloat(kow);

        // Wyświetlenie wczytanej liczby
        document.write("Wczytana liczba: " + a_dana_kow + "<br>");

        // Sprawdzenie znaku liczby
        if (a_dana_kow > 0) {
            document.write("Wczytana liczba jest większa od 0.<br>");
        } else if (a_dana_kow === 0) {
            document.write("Wczytana liczba jest równa 0.<br>");
        } else {
            document.write("Wczytana liczba jest mniejsza od 0.<br>");
        }
    </script>
